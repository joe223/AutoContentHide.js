        var hide1 = new AutoContentHide({
            'LIMIT_PAR_LEN_MIN':1,    // 最低从第一段开始隐藏
            'LIMIT_PAR_LEN_MAX':5,    // 超出第 5 段自动隐藏
            'LIMIT_PROPORTION':1.0,   // 内容长度 / 窗口长度 > 1 时触发隐藏插件开关
            'BREAK':true,             // 段落可以被拆开
            'LIMIT_PAR':10,           // 段落长度超出 10 后可拆开段落（BREAK':true时）
            'BUTTON_CONTENT':"SHOW 1", //展示全文按钮内容
        });

        hide1.AutoHide('hide');    // 传入 id
